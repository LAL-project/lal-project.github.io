145template<
146 class T,
147 std::enable_if_t<
148 std::is_base_of_v<graphs::free_tree, T> ||
149 std::is_base_of_v<graphs::rooted_tree, T>,
150 bool
151 > = true
152>
153std::optional<T> read_head_vector
154(const std::string& filename, bool norm = true, bool check_norm = true)
155noexcept
156{
157 if constexpr (std::is_same_v<graphs::free_tree, T>) {
158 return read_head_vector_free_tree(filename, norm, check_norm);
159 }
160 else if constexpr (std::is_same_v<graphs::rooted_tree, T>) {
161 return read_head_vector_rooted_tree(filename, norm, check_norm);
162 }
163}
