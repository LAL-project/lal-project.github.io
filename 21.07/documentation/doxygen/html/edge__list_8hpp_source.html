168template<
169 class G,
170 std::enable_if_t<std::is_base_of_v<graphs::graph, G>, bool> = true
171>
172std::optional<G>
173read_edge_list
174(const std::string& filename, bool norm = true, bool check_norm = true)
175noexcept
176{
177 if constexpr (std::is_base_of_v<graphs::free_tree, G>) {
178 return read_edge_list_free_tree(filename, norm, check_norm);
179 }
180 else if constexpr (std::is_base_of_v<graphs::rooted_tree, G>) {
181 return read_edge_list_rooted_tree(filename, norm, check_norm);
182 }
183 else if constexpr (std::is_base_of_v<graphs::undirected_graph, G>) {
184 return read_edge_list_undirected_graph(filename, norm, check_norm);
185 }
186 else if constexpr (std::is_base_of_v<graphs::directed_graph, G>) {
187 return read_edge_list_directed_graph(filename, norm, check_norm);
188 }
189}
