58template <class graph_t>
59bool is_node_reachable_from(const graph_t& g, const node source, const node target)
60noexcept
61{
62 BFS<graph_t> bfs(g);
63 bfs.set_terminate
64 ([target](const auto&, const node s) -> bool { return (s == target); });
65 bfs.start_at(source);
66 return bfs.node_was_visited(target);
67}
