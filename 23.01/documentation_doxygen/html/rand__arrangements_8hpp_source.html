101 rand_arrangements(uint64_t n, uint64_t seed = 0) noexcept
102 : m_n(n),
103 m_arr(linear_arrangement(m_n))
104 {
105 if (seed == 0) {
106 std::random_device rd;
107 m_gen = std::mt19937(rd());
108 }
109 else {
110 m_gen = std::mt19937(seed);
111 }
112 m_arr.identity();
113 }
114
116 const linear_arrangement& get_arrangement() noexcept {
117 std::shuffle(m_arr.begin_direct(), m_arr.end_direct(), m_gen);
118 m_arr.update_inverse();
119 return m_arr;
120 }
121
123 const linear_arrangement& yield_arrangement() noexcept {
124 return get_arrangement();
125 }
126
127private:
129 const uint64_t m_n;
131 std::mt19937 m_gen;
137 linear_arrangement m_arr;
