130template <
131 typename result_t, bool second_set_empty,
132 typename iterator_first_t, typename iterator_second_t,
133 class metric,
134 class accumulate_Q, class accumulate_R,
135 class make_average_Q, class make_average_R,
136 class make_average
137>
138[[nodiscard]] result_t one_level_aggregation
139(
140 iterator_first_t bfirst, const iterator_first_t efirst,
141 iterator_second_t bsecond, [[maybe_unused]] const iterator_second_t esecond,
142 metric values,
143 accumulate_Q accQ, accumulate_R accR,
144 make_average_Q avgQ, make_average_R avgR,
145 make_average avg
146)
147noexcept
148{
149 if constexpr (second_set_empty) {
150 // no elements from the second set
151 auto [total_Q, total_R] = values(*bfirst);
152 std::size_t amount = 1;
153
154 ++bfirst;
155 for (; bfirst != efirst; ++bfirst, ++amount) {
156 const auto [Qi, Ri] = values(*bfirst);
157 accQ(total_Q, Qi);
158 accR(total_R, Ri);
159 }
160
161 return avg(avgQ(total_Q, amount), avgR(total_R, amount));
162 }
163 else {
164#if defined DEBUG
165 // ensure same number of elements in both sets
166 assert(std::distance(bfirst, efirst) == std::distance(bsecond, esecond));
167#endif
168
169 auto [total_Q, total_R] = values(*bfirst, *bsecond);
170 std::size_t amount = 1;
171
172 ++bfirst; ++bsecond;
173 for (; bfirst != efirst; ++bfirst, ++bsecond, ++amount) {
174 const auto [Qi, Ri] = values(*bfirst, *bsecond);
175 accQ(total_Q, Qi);
176 accR(total_R, Ri);
177 }
178
179 return avg(avgQ(total_Q, amount), avgR(total_R, amount));
180 }
181}

257[[nodiscard]] result_t two_level_aggregation

261 metric values,
262 combine comb_values, accumulate acc_values,

265noexcept

267 if constexpr (second_set_empty) {
