133template <
134 typename exhaustive_random,
135 typename labelled_unlabelled,
136 class tree_t
137>
138struct tree_generator_type {
139 static_assert(
140 std::is_same_v<exhaustive_random, exhaustive_t> ||
141 std::is_same_v<exhaustive_random, random_t>
142 );
143 static_assert(
144 std::is_same_v<labelled_unlabelled, labelled_t> ||
145 std::is_same_v<labelled_unlabelled, unlabelled_t>
146 );
147
148 typedef
149 std::conditional_t<
150 std::is_same_v<exhaustive_random, exhaustive_t>,
151 // exhaustive types
152 std::conditional_t<
153 std::is_same_v<labelled_unlabelled, labelled_t>,
154 // labelled types
155 std::conditional_t<
156 std::is_base_of_v<graphs::free_tree, tree_t>,
157 all_lab_free_trees,
158 all_lab_rooted_trees
159 >,
160 // unlabelled types
161 std::conditional_t<
162 std::is_base_of_v<graphs::free_tree, tree_t>,
163 all_ulab_free_trees,
164 all_ulab_rooted_trees
165 >
166 >,
167 // random types
168 std::conditional_t<
169 std::is_same_v<labelled_unlabelled, labelled_t>,
170 // labelled types
171 std::conditional_t<
172 std::is_base_of_v<graphs::free_tree, tree_t>,
173 rand_lab_free_trees,
174 rand_lab_rooted_trees
175 >,
176 // unlabelled types
177 std::conditional_t<
178 std::is_base_of_v<graphs::free_tree, tree_t>,
179 rand_ulab_free_trees,
180 rand_ulab_rooted_trees
181 >
182 >
183 >
184 type;
185};
