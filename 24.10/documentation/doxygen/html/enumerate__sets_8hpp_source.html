59template <class graph_t>
60[[nodiscard]] std::vector<edge> set_edges(const graph_t& g) noexcept {
61 std::vector<edge> e(g.get_num_edges());
62 auto vec_it = e.begin();
63 for (iterators::E_iterator<graph_t> e_it(g); not e_it.end(); e_it.next()) {
64 *vec_it++ = e_it.get_edge();
65 }
66 return e;
67}
