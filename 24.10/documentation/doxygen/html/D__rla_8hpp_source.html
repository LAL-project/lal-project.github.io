74[[nodiscard]] inline numeric::rational exp_sum_edge_lengths_rational
75(const graphs::undirected_graph& g)
76noexcept
77{
78 return numeric::rational((g.get_num_nodes() + 1)*g.get_num_edges(), 3);
79}
80
89[[nodiscard]] inline double exp_sum_edge_lengths(const graphs::undirected_graph& g)
90noexcept
91{
92 return exp_sum_edge_lengths_rational(g).to_double();
93}

105[[nodiscard]] inline numeric::rational exp_sum_edge_lengths_rational
