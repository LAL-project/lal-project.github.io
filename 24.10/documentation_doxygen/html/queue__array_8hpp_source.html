68template <typename value_t>
69class queue_array {
70public:
72 void init(const std::size_t n) noexcept {
73 m_queue.resize(n);
74 m_left = 0;
75 m_right = 0;
76 }
77
79 void push(const value_t& v) noexcept {
80#if defined DEBUG
81 assert(not is_full());
82#endif
83 m_queue[m_right++] = v;
84 }
86 void push(value_t&& v) noexcept {
87#if defined DEBUG
88 assert(not is_exhausted());
89#endif
90 m_queue[m_right++] = std::forward<value_t>(v);
91 }
98 [[nodiscard]] value_t&& pop() noexcept {
99#if defined DEBUG
100 assert(m_left < m_queue.size());
101#endif
102 return std::move(m_queue[m_left++]);
103 }
104
106 [[nodiscard]] value_t& front() noexcept {
107#if defined DEBUG
108 assert(not is_exhausted());
109#endif
110 return m_queue[m_left];
111 }
112
114 [[nodiscard]] const value_t& front() const noexcept {
115#if defined DEBUG
116 assert(not is_exhausted());
117#endif
118 return m_queue[m_left];
119 }
121 [[nodiscard]] std::size_t size() const noexcept {
122 return m_right - m_left;
123 }
124
131 void reset() noexcept {
132 m_left = m_right = 0;
133 }
134
142 [[nodiscard]] bool is_exhausted() const noexcept { return m_left == m_queue.size(); }
143
151 [[nodiscard]] bool is_full() const noexcept { return m_right == m_queue.size(); }
152
154 [[nodiscard]] value_t *begin() noexcept { return m_queue.begin() + m_left; }
156 [[nodiscard]] const value_t *begin() const noexcept { return m_queue.begin() + m_left; }
158 [[nodiscard]] value_t *end() noexcept { return m_queue.begin() + m_right; }
160 [[nodiscard]] const value_t *end() const noexcept { return m_queue.begin() + m_right; }
161
162private:
164 array<value_t> m_queue;
165
167 std::size_t m_left;
169 std::size_t m_right;
170};
